聚合函数 分组排序取每组前几条

select id, lottery_id, lottery_date,RANK() OVER(PARTITION BY inshop_lottery_date.lottery_id ORDER BY lottery_date) rank from user_data2.inshop_lottery_date where lottery_id in(2802,2799,2784) and lottery_date >= to_date('2016-06-09 00:00:00','yyyy-mm-dd hh24:mi:ss')
